import imageCompression from 'browser-image-compression';
import { createClient } from '@supabase/supabase-js';

// Tu instancia de Supabase
const supabase = createClient('TU_SUPABASE_URL', 'TU_SUPABASE_ANON_KEY');

async function procesarYSubirImagen(event) {
  const archivoOriginal = event.target.files[0];
  if (!archivoOriginal) return;

  console.log(`Tamaño original: ${(archivoOriginal.size / 1024 / 1024).toFixed(2)} MB`);

  // Configuración de la compresión
  const opciones = {
    maxSizeMB: 0.8,          // El peso máximo que deseas (ej: 800 KB)
    maxWidthOrHeight: 1280, // Redimensiona si la foto es gigante (4K, etc.)
    useWebWorker: true,     // Corre en segundo plano para no trabar la UI
    fileType: 'image/webp'  // ¡Truco pro! Convierte a WebP para que pese aún menos
  };

  try {
    // 1. Comprimir la imagen en el cliente
    const archivoComprimido = await imageCompression(archivoOriginal, opciones);
    console.log(`Tamaño final: ${(archivoComprimido.size / 1024 / 1024).toFixed(2)} MB`);

    // 2. Generar un nombre único para evitar duplicados en Supabase
    const nombreArchivo = `${Date.now()}_foto.webp`;

    // 3. Subir directamente a tu bucket de Supabase
    const { data, error } = await supabase.storage
      .from('tus-imagenes') // Nombre de tu bucket
      .upload(nombreArchivo, archivoComprimido, {
        contentType: 'image/webp', // Asegura el tipo MIME correcto
      });

    if (error) throw error;

    alert('¡Imagen optimizada y subida con éxito!');
    console.log('Datos de la subida:', data);

  } catch (error) {
    console.error('Error en el proceso:', error);
  }
}
